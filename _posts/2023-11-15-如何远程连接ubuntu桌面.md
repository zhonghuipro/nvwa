```shell
apt update
apt install ubuntu-desktop
apt install xrdp
adduser xrdp ssl-cert //将xrdp 用户加入到 ssl-cert用户组中，登录时用到
passwd xrdp //设置xrdp的登录密码
systemctl restart xrdp
```
